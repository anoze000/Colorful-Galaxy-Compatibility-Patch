planet_event = {
	id = lv_event_popups.1
	hide_window = yes
	
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				NOR = {
					has_global_flag = lv_events_hidden
					has_global_flag = lv_brood_parasite_hidden
					has_global_flag = lv_all_disabled
					has_global_flag = lv_rape_disabled
				}
			}
			random_list = {
				25 = {
					planet_event = { id = lv_event_popups.2 }
				}
				25 = {
					planet_event = { id = lv_event_popups.3 }

				25 = {
					planet_event = { id = lv_event_popups.4 }
				}
				25 = {
					planet_event = { id = lv_event_popups.5 }
				}
			}
		}
	}
}

planet_event = {	# Brood Parasites 1
	id = lv_event_popups.2
	title = "Brood Parasites Breeding"
	desc = "lv_brood_parasite_inform.desc"
	diplomatic = yes
	custom_gui = "lv_h_window"
	custom_gui_option = "lv_h_option"
	show_sound = event_swarm_started
	location = ROOT
	trackable = yes
	
	picture_event_data = {
		room = evt_lv_brood_parasite_1_room
	}

	is_triggered_only = yes

	option = {
		name = lv_ep_brood_parasite.name.1
		custom_tooltip = lv_ep_brood_parasite.name.1.tooltip
		trigger = {
			owner = { 
				OR = {
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
				} 
			}
		}
	}
	option = {
		name = lv_ep_brood_parasite.name.2
		custom_tooltip = lv_ep_brood_parasite.name.2.tooltip
		trigger = {
			owner = {
				AND = {
					OR = {
						has_ethic = ethic_xenophobe
						has_ethic = ethic_fanatic_xenophobe
					}
					NOT = {
						owner_species = {
							has_trait = trait_lv_brood_parasite
						}
					}
				}
			}
		}
	}
	option = {
		name = lv_ep_brood_parasite.name.3
