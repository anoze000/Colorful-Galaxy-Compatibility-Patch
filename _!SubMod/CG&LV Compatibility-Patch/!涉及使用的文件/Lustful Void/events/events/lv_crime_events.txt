# Trafficking #

planet_event = {
	id = lv_crime_events.1
	title = lv_crime_events.1.name
	desc =  lv_crime_events.1.desc
	diplomatic = yes
	custom_gui = "lv_h_window"
	custom_gui_option = "lv_h_option"
	show_sound = event_criminal_activity
	location = ROOT
	trackable = yes
	
	picture_event_data = {
		room = evt_lv_crime_trafficking_1_room
	}
	
	is_triggered_only = yes
	
	pre_triggers = {
		has_owner = yes
	}
	
	trigger = {
		has_modifier = criminal_underworld
	}
	
	immediate = {
		set_timed_planet_flag = {
			flag = lv_crime_trafficking
			days = 1800
		}
	}
	
	option = {
		name = lv_crime_events.1.a
		add_modifier = { modifier = lv_crime_trafficking days = 1800 }
	}
}
