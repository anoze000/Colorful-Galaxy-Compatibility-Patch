planet_event = {
	id = lv_game_start.3
	hide_window = yes
	
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				owner = {
					OR = {
						has_origin = origin_lv_parallel_evolution
						has_origin = origin_lv_sadomasochistic_evolution
						has_origin = origin_lv_symbiotic_evolution
					} 
				}
			}
			
			if = {
				limit = { owner = { has_origin = origin_lv_parallel_evolution } }
		
				last_created_species = {
					set_species_flag = lv_parallel_species@root.owner
					set_citizenship_type = {
						country = root.owner
						type = citizenship_slavery
					}
				}
				while = {
					count = 12
					create_pop = {
						species = last_created_species
						ethos = random
					}
				}
			}
			
			if = {
				limit = { owner = { has_origin = origin_lv_sadomasochistic_evolution } }
		
				last_created_species = {
					set_species_flag = lv_sadomasochistic_species@root.owner
					set_citizenship_type = {
						country = root.owner
						type = citizenship_slavery
					}
				}
				while = {
					count = 12
					create_pop = {
						species = last_created_species
						ethos = { ethic = ethic_authoritarian }
					}
				}
			}
			
			if = {
				limit = { owner = { has_origin = origin_lv_symbiotic_evolution } }
				
				last_created_species = {
					set_citizenship_type = {
						country = root.owner
						type = citizenship_full
					}
					set_living_standard = {
						country = root.owner
						type = living_standard_hive_mind
					}
				}
				
				while = {
					count = 16
					create_pop = {
						species = last_created_species
						ethos = random
					}
					last_created_pop = {
						clear_ethos = yes
					}
				}
			}
		}
	}
}
