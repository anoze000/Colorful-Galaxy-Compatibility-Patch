# Sadistic/Rapist Celebration #

event = {
	id = lv_fleet_events.1
	title = lv_fleet_events.1.name
	desc = lv_fleet_events.1.desc
	diplomatic = yes
	custom_gui = "lv_h_window"
	custom_gui_option = "lv_h_option"
	show_sound = event_bad_omen
	location = fromfrom
	trackable = yes
	
	picture_event_data = {
		room = evt_lv_sadistic_rapist_celebration_1_room
	}
	
	is_triggered_only = yes
	
	trigger = {
		NOR = {
			has_global_flag = lv_all_disabled
			has_global_flag = lv_rape_disabled
		}
		fromfrom = {
			exists = leader
			leader = {
				OR = {
					has_trait = leader_trait_lv_tormentor
					has_trait = leader_trait_commander_lv_rapist
				}
			}
			NOT = {
				has_fleet_flag = lv_had_fleet_event
			}
		}
		fromfromfrom = {
			lv_is_not_a_bad_dwagon = yes
		}
	}
	
	immediate = {
		fromfrom = {
			leader = { save_event_target_as = officer }
			solar_system = { save_event_target_as = system }
			set_timed_fleet_flag = { flag = lv_had_fleet_event days = 360 }
		}
	}
	
	option = {
		name = lv_fleet_events.1.a
		trigger = {
			this = {
				NOR = {
					has_ethic = ethic_pacifist
					has_ethic = ethic_fanatic_pacifist
				}
			}
		}
		add_monthly_resource_mult = {
            resource = influence
            value = @tier1influencereward
            min = @tier1influencemin
            max = @tier1influencemax
		}
	}
	
	option = {
		name = lv_fleet_events.1.b
		trigger = {
			this = {
				OR = {
					has_ethic = ethic_authoritarian
					has_ethic = ethic_fanatic_authoritarian
				}
			}
		}
		add_monthly_resource_mult = {
            resource = influence
            value = @tier2influencereward
            min = @tier2influencemin
            max = @tier2influencemax
        }
	}
	
	option = {
		name = lv_fleet_events.1.c
		trigger = {
			this = {
				OR = {
					has_ethic = ethic_militarist
					has_ethic = ethic_fanatic_militarist
					AND = {
						OR = {
							has_ethic = ethic_xenophobe
							has_ethic = ethic_fanatic_xenophobe
						}
						event_target:officer.species = {
							NOT = {
								is_same_species = event_target:defeated_officer.species
							}
						}
					}
				}
			}
		}
		add_monthly_resource_mult = {
            resource = influence
            value = @tier1influencereward
            min = @tier1influencemin
            max = @tier1influencemax
        }
		add_monthly_resource_mult = {
            resource = unity
            value = @tier1unityreward
            min = @tier1unitymin
            max = @tier1unitymax
        }
	}
	
	option = {
		name = lv_fleet_events.1.d
		trigger = {
			this = {
				OR = {
					has_ethic = ethic_pacifist
					has_ethic = ethic_fanatic_pacifist
				}
			}
		}
		add_resource = { influence = -10 }
	}
}
