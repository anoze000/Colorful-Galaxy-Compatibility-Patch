# Paternal Chip
trait_lv_paternal_chip_cyborg = {
	cost = {
		base = 3
		modifier = {
			add = -3
			has_country_flag = lv_cheat_mode
		}
		modifier = {
			mult = 0
			owner = { has_valid_civic = civic_augmentation_bazaars }
		}
	}
	allowed_archetypes = { BIOLOGICAL LITHOID }
	custom_tooltip = trait_lv_paternal_chip_cyborg_effect
	
	opposites = {
		trait_lv_maternal_chip_cyborg
	}
	
	initial = no
	species_potential_add = {
		can_add_cybernetic_traits = yes
	}
	species_possible_remove = {
		can_remove_cybernetic_traits = yes
	}
	randomized = no
	category = cyborg
	
	modifier = {
		# Lead by example
		army_damage_mult = 0.2
		army_attack_morale_mult = 0.1
		# Hard work pays off
		planet_jobs_energy_produces_mult = 0.15
		planet_jobs_minerals_produces_mult = 0.25
	}
	
	slave_cost = {
		energy = 150
	}
}
