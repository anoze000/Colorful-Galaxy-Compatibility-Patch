planet_event = {
	id = MGHG_game_start.03
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		owner = {
			is_monster_girl = yes
		}
	}

	immediate = {
		owner = {
			add_resource = {
				lust_crystal = 1000
			}
		}
		while = {
			count = 3
			if = {
				limit = {
					free_district_slots > 0
				}
				if = {
					limit = {
						is_planet_class = pc_habitat
					}
					add_district = district_hab_red_lights
				}
				else = {
					add_district = district_red_lights
				}
			}
			else = {
				if = {
					limit = {
						is_planet_class = pc_habitat
					}
					remove_district = district_hab_housing
					add_district = district_hab_red_lights
				}
				if = {
					limit = {
						owner = {
							is_regular_empire = yes
						}
					}
					remove_district = district_city
					add_district = district_red_lights
				}
				else_if = {
					limit = {
						owner = {
							is_hive_empire = yes
						}
					}
					remove_district = district_hive
					add_district = district_red_lights
				}
				else_if = {
					limit = {
						owner = {
							is_machine_empire = yes
						}
					}
					remove_district = district_nexus
					add_district = district_red_lights
				}
			}
		}
		owner = {
			give_technology = { tech = "tech_non_conservation_sexual_energy_theorem" message = no }
			give_technology = { tech = "tech_sexual_energy_engine" message = no }
		}

		add_building = building_sexual_energy_resource_center_1

	}
