# Pleasure Suit #
lv_add_suit = {
	if = {
		limit = {
			NOR = {
				has_trait = leader_trait_lv_pleasure_suit_addict
				has_trait = leader_trait_lv_pleasure_suit_addict_2
				has_trait = leader_trait_lv_pleasure_suit_addict_3

# Cow Transformation #
# root/this = planet
lv_milk_cow_transform_effect = {
	every_owned_pop = {
		limit = {
			is_robot_pop = no
			OR = {
				NOT = { pop_has_trait = trait_lv_all_male }
				pop_has_trait = trait_lv_femboy05
			}
		}
		if = {
			limit = { 
				NOR = {
					has_pop_flag = lv_milk_cow_transform_1
					has_pop_flag = lv_milk_cow_transform_2
					has_pop_flag = lv_milk_cow_transform_3
					has_pop_flag = lv_milk_cow_transform_4
					has_modifier = lv_milk_cow_transform_complete
				}
			}
			set_pop_flag = lv_milk_cow_transform_1
			lv_mct_1_effect = yes
		}
		else_if = {
			limit = { has_pop_flag = lv_milk_cow_transform_1 }
			remove_pop_flag = lv_milk_cow_transform_1
			set_pop_flag = lv_milk_cow_transform_2
			lv_mct_2_effect = yes
		}
		else_if = {
			limit = { has_pop_flag = lv_milk_cow_transform_2 }
			remove_pop_flag = lv_milk_cow_transform_2
			set_pop_flag = lv_milk_cow_transform_3
			lv_mct_3_effect = yes
		}
		else_if = {
			limit = { has_pop_flag = lv_milk_cow_transform_3 }
			remove_pop_flag = lv_milk_cow_transform_3
			set_pop_flag = lv_milk_cow_transform_4
			lv_mct_4_effect = yes
		}
		else_if = {
			limit = { has_pop_flag = lv_milk_cow_transform_4 }
			remove_pop_flag = lv_milk_cow_transform_4
			add_modifier = {
				modifier = lv_milk_cow_transform_complete
				days = -1
			}
			lv_mct_5_effect = yes
			owner = {
				add_monthly_resource_mult = {
					resource = society_research
					value = 1
					min = 1
					max = 100
				}
			}
		}
	}
}
