#Set to indicate overwrites are active
set_downstairs_override = {
	always = no
}

#Everything below operates at species scope

set_downstairs_male = {
	OR = {
		has_trait = trait_lv_all_male
		has_trait = trait_CD
	}
}

set_downstairs_female = {
	OR = {
		has_trait = trait_lv_all_female
		has_trait = trait_suc_parthenogenesis
		AND = {
			OR = {
				has_trait = trait_slime
				has_trait = trait_SLIME_robot
			}
			is_futa_monster_girl = no
		}
	}	
}

set_downstairs_herm = {
	OR = {
		has_trait = trait_lv_futanari
		is_futa_monster_girl = yes
	}
}

	has_trait = trait_lv_androgynous
}

	has_trait = trait_lv_excess_estrogen
}

shift_female_to_andro = {
