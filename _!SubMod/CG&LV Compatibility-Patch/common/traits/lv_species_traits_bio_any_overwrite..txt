#####################################
###     Biological/Any Traits     ###
#####################################
#
# Note: All traits were made with the intention of adding events/other things relating to them
# eventually, even if not stated on the traits themselves. Also, all traits can and probably
# will change again! These are not final effects or values at all.
#
# The potential_crossbreeding_chance modifier is used for base game crossbreeding compatibility 
# only.
#
# The following categories are more for organizational purposes than anything else:
# Psychological  - Mentality or mind altering effects.
# Physical       - Body parts or other physical attributes.

# Submissive
trait_lv_submissive = {
	cost = {
		base = 2
		modifier = {
			add = -2
			has_country_flag = lv_cheat_mode
		}
	}
	potential_crossbreeding_chance = 1.0
	allowed_archetypes = { BIOLOGICAL CG_species_archetype PRESAPIENT LITHOID }
	custom_tooltip = trait_lv_submissive_effect
	
	species_possible_remove = {
		exists = from
		from = {
			OR = {
				can_remove_beneficial_genetic_traits = yes
				has_country_flag = lv_cheat_mode
			}
		}
	}
	species_possible_merge_remove = {
		always = yes
	}
	
	opposites = {
		trait_lv_dominant
		trait_unruly
		trait_sedentary
		trait_deviants
		trait_decadent
		trait_hive_mind
		trait_machine_unit
		trait_enslavement_resistance
		trait_forbidden_lust
		trait_virtuous
	}
	
	triggered_pop_modifier = {
		potential = {
			exists = planet
			planet = {
				any_owned_pop = {
					pop_has_trait = trait_lv_dominant
				}
			}
		}
		# Happy to share a planet with doms
		pop_amenities_usage_mult = -0.05
		pop_housing_usage_mult = -0.1
		species_empire_size_mult = -0.25
	}
	triggered_pop_modifier = {
		potential = {
			exists = owner
			owner = {
				ruler = {
					OR = {
						has_trait = leader_trait_lv_highly_dominant
						has_trait = leader_trait_lv_highly_dominant_2
						has_trait = leader_trait_lv_highly_dominant_3
					}
				}
			}
		}
		# Adores dominant rulers
		pop_happiness = 0.15
		pop_resettlement_cost_mult = -1
	}
	
	modifier = {
		# Hates being in charge
		army_morale = -0.75
		planet_jobs_ruler_produces_mult = -0.05
		planet_jobs_worker_produces_mult = 0.05
		planet_jobs_slave_produces_mult = 0.1
		pop_demotion_time_mult = -0.5
	}

	slave_cost = {
		energy = 2000
	}
}
