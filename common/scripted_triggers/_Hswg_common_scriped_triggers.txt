is_canbe_sex_slave = {
	OR = {
		species = {
			OR = {
				#DC物种
				is_archetype = Magical_Bitch
				is_archetype = Magical_Girl
				is_archetype = Lust_Girl
				#DC附属物种
				is_archetype = MONSTER
				is_archetype = PARASITE
				#联动物种
				is_archetype = NYTO
				is_archetype = LILY
				is_archetype = Dolls
			}
		}
		AND = {
			species = {
				OR = {
					is_archetype = LOVEMACHINE
					is_archetype = YGWAWA
				}
			}
			has_trait = trait_cyber_RBQ
		}
		#舰r相关
		AND = {
			species = {
				OR = {
					is_archetype = WARSHIPGIRL
					is_archetype = SHENHAI
					is_archetype = UNITED_FLEET
					is_archetype = NANITES_CROP
				}
			}
			OR = {
				has_trait = trait_DC_ethic_carnalism
				has_trait = trait_DC_ethic_asceticism
			}			
		}
		#星舰相关
		AND = {
			species = {
				is_archetype = STARSHIPGIRLS
			}
			OR = {
				has_trait = trait_DC_ethic_carnalism
				has_trait = trait_DC_ethic_asceticism
			}		
		}
	}
}
