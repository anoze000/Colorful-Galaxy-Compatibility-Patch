ap_power_of_heaven = {
	on_enabled = {
		custom_tooltip = "power_of_heaven_tooltip"
		hidden_effect = { set_country_flag = str_nirvana_awakened }
		add_research_option = tech_celestial_lance
		add_research_option = tech_celestial_beam
		add_research_option = tech_celestial_laser
		add_research_option = tech_nirvana_shield
		add_research_option = tech_nirvana_armor
		add_research_option = tech_nirvana_reactor
		give_technology = { tech = tech_fallen_throne }
		give_technology = { tech = tech_gateway_activation message = no }
		give_technology = { tech = tech_gateway_construction message = no }
	}
	modifier = {
		megastructure_build_speed_mult = 1
		country_resource_max_add = 10000000
		country_produces_mult = 0.5
		modify_species_cost_mult = -1
		BIOLOGICAL_species_trait_points_add = 5
		WARSHIPGIRL_species_trait_points_add = 5
		SHENHAI_species_trait_points_add = 5
		WHORE_species_trait_points_add = 5
		INNOCENTSHOJO_species_trait_points_add = 5
		TENTACLE_species_trait_points_add = 5
		STARSHIPGIRLS_species_trait_points_add = 5
		TOUHOUGIRLS_species_trait_points_add = 5
		Dolls_species_trait_points_add = 5
		UNITED_FLEET_species_trait_points_add = 5
		NYTO_species_trait_points_add = 5
	}
	possible = {
		custom_tooltip = {
			fail_text = "synthetic_empire_synthetic_ascension"
			NOT = { has_trait = trait_mechanical }
		}
		custom_tooltip = {
			fail_text = "requires_technology_dragon_rise"
			has_technology = tech_dragon_rise
		}
		custom_tooltip = {
			fail_text = "requires_technology_light_of_civilization"
			has_technology = tech_light_of_civilization
		}
		custom_tooltip = {
			fail_text = "hive_mind_biological_ascension_only2"
			NOT = { has_authority = auth_hive_mind }
		}
		custom_tooltip = {
			fail_text = "requires_ascension_perks_1"
			num_ascension_perks > 0
		}
		custom_tooltip = {
			fail_text = "requires_not_ap_the_flesh_is_weak"
			NOT = { has_ascension_perk = ap_the_flesh_is_weak }
		}
		custom_tooltip = {
			fail_text = "requires_mod_ap_mind_over_matter_str"
			OR = {
				has_ascension_perk = ap_mind_over_matter
				has_ascension_perk = ap_toho4
				has_ascension_perk = ap_hm4
				has_ascension_perk = ap_mist
				has_ascension_perk = ap_shenhai_silent
				has_ascension_perk = ap_shenhai_icebreaking
				has_ascension_perk = ap_galactic_empire
				has_ascension_perk = ap_lustgod_tentacle_symbiosis
				has_ascension_perk = ap_TARDIS
				has_ascension_perk = ap_project_s
				has_ascension_perk = ap_uf_2
				has_ascension_perk = ap_become_the_gray
				has_ascension_perk = ap_GF_nyto_resurrection
			}
		}
	}
	potential = {
		is_nirvana_empire_dragon = yes
		NOR = {
			has_ascension_perk = ap_power_of_heaven
			has_authority = "auth_machine_intelligence"
		}
		is_mechanical_empire = no
	}
	ai_weight = { factor = 100 }
}

ap_god_of_heaven = {
	on_enabled = {
		custom_tooltip = "god_of_heaven_tooltip"
		hidden_effect = { set_country_flag = str_nirvana_final_awakened }
		add_research_option = tech_god_of_nirvana
	}
	modifier = {
		ships_upkeep_mult = -1
		ship_windup_mult = -1
		ship_winddown_mult = -1
		ship_ftl_jumpdrive_range_mult = 1
		ship_jumpdrive_cooldown_mult = -1
		megastructure_build_speed_mult = 1
		country_resource_max_add = 10000000
		country_produces_mult = 0.5
		command_limit_mult = 1
	}
	possible = {
		custom_tooltip = {
			fail_text = "synthetic_empire_synthetic_ascension"
			NOT = { has_trait = trait_mechanical }
		}
		custom_tooltip = {
			fail_text = "hive_mind_biological_ascension_only2"
			NOT = { has_authority = auth_hive_mind }
		}
		custom_tooltip = {
			fail_text = "requires_ascension_perks_1"
			num_ascension_perks > 0
		}
		custom_tooltip = {
			fail_text = "mod_requires_ap_power_of_heaven"
			has_ascension_perk = ap_power_of_heaven
		}
		custom_tooltip = {
			fail_text = "mod_requires_hundred_years"
			OR = {
				has_origin = origin_kuat_apocalyptic
				AND = { years_passed >= 50 is_ai = yes }
				AND = { years_passed >= 75 is_ai = no }
			}
		}
	}
	potential = {
		is_nirvana_empire_dragon = yes
		NOT = { has_ascension_perk = ap_god_of_heaven }
	}
	ai_weight = {
		factor = 1
		modifier = {
			factor = 100
			OR = {
				has_global_flag = wg_crisis_insane
				has_global_flag = afk_crisis_real
			}
		}
		modifier = { factor = 5 end_game_years_passed >= 5 }
	}
}
