### Federations Scripted Triggers
nirvana_losing_defensive_war = {
	optimize_memory
	any_war = {
		any_defender = {
			is_same_value = root
		}
		any_attacker = {
			OR = {
				relative_power = {
					who = root
					category = fleet
					value = overwhelming
				}
				relative_power = {
					who = root
					category = fleet
					value = superior
				}
			}
		}
		defender_war_exhaustion > 0.3
		attacker_war_exhaustion < 0.7
	}
}
