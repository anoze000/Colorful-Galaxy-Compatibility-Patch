#圣光解封
ap_holy_light_remove_seal = {
	on_enabled = {
		custom_tooltip = "holy_light_remove_seal_text"
		hidden_effect = { country_event = { id = mind_break.1 } }
	}
	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_1"
			OR = {
				num_ascension_perks > 0
				is_ai = yes
			}
		}
		OR = {
			has_trait = trait_forbidden_lust
			has_origin = origin_INN_destruction_light
		}
		NOR = {
			has_ascension_perk = ap_the_flesh_is_weak
			has_ascension_perk = ap_synthetic_evolution
		}
		custom_tooltip = {
			fail_text = "requires_pure_policy_flag"
			OR = {
				has_policy_flag = lust_forbidden
				is_ai = yes
			}
		}
		custom_tooltip = {
			fail_text = "requires_tech_holylight_deep_research"
			OR = {
				has_technology = tech_holylight_deep_research
				is_ai = yes
			}
		}
	}
	potential = {
		is_pure_empire = yes
		OR = {
			is_mechanical_empire = no
			is_dc_cor_archetype = yes
		}
		NOT = { has_ascension_perk = ap_holy_light_remove_seal }
	}
	modifier = {
		planet_jobs_holylight_crystal_produces_mult = 0.1
	}
	ai_weight = { factor = 500 }
}
