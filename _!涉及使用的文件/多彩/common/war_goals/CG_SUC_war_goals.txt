# 战争目标：异种改造
wg_suc_trans_uglypop = {
	casus_belli = cb_suc_trans_uglypop
	hide_if_no_cb = yes
	total_war = no
	galactic_empire_joins_defender = {
		from = { is_galactic_emperor = yes }
		imperial_authority >= 100
	}
	cede_claims = no
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}
	surrender_acceptance = -100
	war_exhaustion = 0.5
	potential = { always = yes }
	possible = {
		custom_tooltip = {
			fail_text = cannot_use_trans_uglypop
			from = {
				any_owned_pop = { is_ugly_pop = yes }
			}
		}
	}
	on_accept = {
		custom_tooltip = wg_suc_trans_uglypop
		hidden_effect = {
			from = {
				every_owned_planet = {
					limit = {
						any_owned_pop = { is_ugly_pop = yes }
						NOT = { has_modifier = CG_decision_alien_transforming_modifer }
					}
					planet_event = { id = CG_suc_decisions.10 }
				}
			}
		}
	}
	on_status_quo = {
		custom_tooltip = wg_suc_trans_uglypop
		hidden_effect = {
			from = {
				every_owned_planet = {
					limit = {
						any_owned_pop = { is_ugly_pop = yes }
						NOT = { has_modifier = CG_decision_alien_transforming_modifer }
					}
					planet_event = { id = CG_suc_decisions.10 }
				}
			}
		}
	}
	ai_weight = { weight = 0 }
}
