ap_engineered_evolution = {
	on_enabled = {
		custom_tooltip = "engineered_evolution_tooltip"
		if = {
			limit = {
				has_origin = origin_clone_army
				OR = {
					has_trait = trait_clone_soldier_infertile
					has_trait = trait_clone_soldier_infertile_full_potential
				}
			}
			custom_tooltip = clone_soldier_locked_genes
		}
		# 腐化主脑
		if = {
			limit = {
				has_origin = origin_corrupt_brain
			}
			custom_tooltip = can_cure_corrupt_brain
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = "synthetic_empire_biological_ascension"
			NOT = { has_trait = trait_mechanical }
		}
		custom_tooltip = {
			fail_text = "requires_ascension_perks_2"
			num_ascension_perks > 1
		}
		if = {
			limit = {
				NOT = { has_global_flag = time_ascension_paths }
			}
			custom_tooltip = {
				fail_text = "requires_not_ap_the_flesh_is_weak"
				NOT = { has_ascension_perk = ap_the_flesh_is_weak }
			}
			custom_tooltip = {
				fail_text = "requires_not_ap_organo_machine_interfacing"
				NOT = { has_ascension_perk = ap_organo_machine_interfacing }
			}
			custom_tooltip = {
				fail_text = "requires_not_ap_mind_over_matter"
				NOT = { has_ascension_perk = ap_mind_over_matter }
			}
		}
		custom_tooltip = {
			fail_text = "requires_not_ap_synthetic_evolution"
			NOT = { has_ascension_perk = ap_synthetic_evolution }
		}
		# 魔女之旅
		custom_tooltip = {
			fail_text = "requires_not_ap_matters_all_have_minds"
			NOR = { 
				has_ascension_perk = ap_matters_all_have_minds
				has_tradition = tr_objects_wv_adopt 
			}
		}	
	}

	potential = {
		ap_engineered_evolution_potential = yes
	}

	ai_weight = {
		factor = @ascension_path_base_weight_low
		modifier = {
			factor = @ascension_path_synergy_weight
			OR = {
				has_authority = auth_hive_mind
				has_engineered_evolution_synergy = yes
			}
		}
	}
}
