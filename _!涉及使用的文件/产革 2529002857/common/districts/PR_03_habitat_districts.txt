PR_district_HAB_housing = {
	base_buildtime = @hab_time
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes

		uses_district_set = habitat
		has_planet_flag = PR_flg_plnt_VALID
		NOR = {
			has_planet_flag = toxic_knights_habitat
			has_building = building_order_castle
		}
	}
	potential = {
		uses_district_set = habitat
		has_planet_flag = PR_flg_plnt_VALID

			exists = owner

				has_planet_flag = toxic_knights_habitat
				has_building = building_order_castle
			}
		}

		PR_district_HAB_housing	# PR居住站

		category = planet_districts
		cost = {
			minerals = @hab_cost
		}

			energy = @hab_maintenance
			alloys = @low_hab_maintenance
		}
	}

		planet_housing_add = 6
		planet_max_buildings_add = @PR_bldn_HAB_city
	}

		potential = { exists = owner	owner = { has_active_tradition = tr_prosperity_void_works } }
		planet_housing_add = 1.5
	}

	###	script = districts/habitat_housing_job_swaps	############################################################
	##	普通|默认

		potential = { exists = owner	owner = { is_gestalt = no }		has_farming_designation = no }
		planet_housing_add = 1
		trade_value_add = 1

	}

		trigger = { exists = owner	owner = { is_gestalt = no }		has_farming_designation = no }
		text = job_clerk_effect_desc
	}
	#	普通|商业
	triggered_planet_modifier = {
		potential = { exists = owner	owner = { is_gestalt = no }		colony_type = col_habitat_trade }
		trade_value_add = 2	#4-2
		planet_amenities_add = -2
		job_clerk_add = -2	#base=4
		job_trader_add = 2
	}
	triggered_desc = {
		trigger = { exists = owner	owner = { is_gestalt = no }		colony_type = col_habitat_trade }
		text = job_trader_effect_desc
	}
	##	格式塔|默认
	triggered_planet_modifier = {
		potential = { exists = owner	owner = { is_gestalt = yes }		has_farming_designation = no }
		planet_housing_add = 1
		planet_amenities_no_happiness_add = 1
		job_maintenance_drone_add = 1
		mult = owner.value:PR_sv_cntr_HAB_void_dweller_jobs
	}
	triggered_desc = {
		trigger = { exists = owner	owner = { is_gestalt = yes }		has_farming_designation = no }
		text = job_maintenance_drone_effect_desc
	}
	##	全部|农业
